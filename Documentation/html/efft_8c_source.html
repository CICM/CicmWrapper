49 {
50  float a0 = 0.5;
51  for(int i = 0; i < size; i++)
52  {
53  if(i <= ((a0 * (double)(size - 1)) / 2.))
54  window[i] = 0.5 * ( 1. + cos(EPD_PI * ( ((2 * (double)i) / (a0 * (double)(size - 1))) - 1.)));
55  else if(i > ((a0 * (double)(size - 1)) / 2.) && i <= ((double)(size - 1) * (1. - ( a0 / 2.))))
56  window[i] = 1.;
57  else
58  window[i] = 0.5 * ( 1. + cos(EPD_PI * ( ((2 *(double)i) / (a0 * (double)(size - 1))) + 1. - (2. / a0))));
59  }
60 }
