131  {
132  if(c->c_attr[i].name == gensym(attrname))
133  {
134  if(gensym(style) == gensym("checkbutton") || gensym(style) == gensym("onoff"))
135  {
136  c->c_attr[i].style = gensym("checkbutton");
137  }
138  else if(gensym(style) == gensym("rgb") || gensym(style) == gensym("rgba") || gensym(style) == gensym("color"))
139  {
140  c->c_attr[i].style = gensym("color");
141  }
142  else if(gensym(style) == gensym("spinbox") || gensym(style) == gensym("number"))
143  {
144  c->c_attr[i].style = gensym("number");
145  }
146  else if(gensym(style) == gensym("menu"))

149  }

168 void eclass_attr_itemlist(t_eclass* c, char* attrname, long flags, char* list)
169 {
170  int i, j = 0;
171  char* pch;
172  int size = 0;
173  for(i = 0; i < c->c_nattr; i++)
174  {
175  if(c->c_attr[i].name == gensym(attrname))
176  {
177  pch = strtok(gensym(list)->s_name," ,");
178 
179  while(pch != NULL)
180  {
181  pch = strtok(NULL, " ,");
182  size++;
183  }
184  if(size > 0)
185  {
186  if(c->c_attr[i].itemssize)
187  {
188  c->c_attr[i].itemslist = (t_symbol **)realloc(c->c_attr[i].itemslist, size * sizeof(t_symbol *));
189  if(c->c_attr[i].itemslist)
190  c->c_attr[i].itemssize = size;
191  }
192  else
193  {
194  c->c_attr[i].itemslist = (t_symbol **)calloc(size, sizeof(t_symbol *));
195  if(c->c_attr[i].itemslist)
196  c->c_attr[i].itemssize = size;
197  }
198  if(c->c_attr[i].itemssize)
199  {
200  pch = strtok(gensym(list)->s_name," ,");
201  while(pch != NULL)
202  {
203  c->c_attr[i].itemslist[j] = gensym(pch);
204  pch = strtok(NULL, " ,");
205  j++;
206  }
207  }
208 
209  }
210  else
211  {
212  if(c->c_attr[i].itemssize)
213  free(c->c_attr[i].itemslist);
214  c->c_attr[i].itemssize = 0;
215  }
216  return ;
217  }
218  }
219 }
