66 void egraphics_paint(t_elayer *g, int filled, int preserved)
67 {
68  int i, index;
69  if(g->e_new_objects.e_type != E_GOBJ_INVALID)
70  {
71  g->e_objects = (t_egobj *)realloc(g->e_objects, (g->e_number_objects + 1) * sizeof(t_egobj));
72  if(g->e_objects)
73  {
74  index = (int)g->e_number_objects;
75  g->e_number_objects++;
76  if (filled)
77  {
78  g->e_objects[index].e_filled = 1;
79  }
80  else
81  {
82  g->e_objects[index].e_filled = 0;
83  }
84  g->e_objects[index].e_type = g->e_new_objects.e_type;
85  g->e_objects[index].e_roundness = g->e_new_objects.e_roundness;
86  g->e_objects[index].e_npoints = g->e_new_objects.e_npoints;
87  g->e_objects[index].e_points = (t_pt*)calloc(g->e_objects[index].e_npoints, sizeof(t_pt));
88  if(!g->e_objects[index].e_points)
89  {
90  g->e_objects[index].e_type = E_GOBJ_INVALID;
91  return;
92  }
93  for(i = 0; i < g->e_objects[index].e_npoints; i++)
94  {
95  g->e_objects[index].e_points[i] = g->e_new_objects.e_points[i];
96  }
97  g->e_objects[index].e_roundness = g->e_new_objects.e_roundness;
98 
99  g->e_objects[index].e_color = g->e_color;
100  g->e_objects[index].e_width = g->e_line_width;
101  g->e_objects[index].e_text = g->e_new_objects.e_text;
102 
103  egraphics_apply_matrix(g, g->e_objects+index);
104  if(!preserved)
105  {
106  g->e_new_objects.e_roundness = 0;
107  }
108  }
109  }
110 }
