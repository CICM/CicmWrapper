32  \ @param name The eclass name
33  \ @param newmethod The new method
34  \ @param freemethod The free method
35  \ @param size The size of the eclass

41 t_eclass* eclass_new(char *name, method newmethod, method freemethod, size_t size, int flags, t_atomtype arg1, int arg2)
42 {
43  t_class *pd = class_new(gensym(name), (t_newmethod)newmethod, (t_method)freemethod, size, flags, arg1, arg2);
44  t_eclass* c;
45  c = (t_eclass *)resizebytes(pd, sizeof(*pd), sizeof(*c));
46  c->c_nattr = 0;
47  c->c_dsp = 0;
48  c->c_box = 0;
49  c->c_attr = (t_eattr *)malloc(sizeof(t_eattr));
50  eproxy_setup(c);
51  erouter_setup();
52  return c;
53 }
