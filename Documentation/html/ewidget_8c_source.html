61 void ewidget_getrect(t_gobj *z, t_glist *glist, int *xp1, int *yp1, int *xp2, int *yp2)
62 {
63  t_ebox *x = (t_ebox *)z;
64  *xp1 = text_xpix(&x->b_obj.o_obj, glist);
65  *yp1 = text_ypix(&x->b_obj.o_obj, glist) - (int)(x->b_boxparameters.d_borderthickness);
66  *xp2 = text_xpix(&x->b_obj.o_obj, glist) + (int)x->b_rect.width + (int)(x->b_boxparameters.d_borderthickness);

70 void ewidget_vis(t_gobj *z, t_glist *glist, int vis)
71 {
72  t_ebox* x = (t_ebox *)z;
73  t_eclass* c = (t_eclass *)x->b_obj.o_obj.te_g.g_pd;
74 
75  if(vis)
76  {
77  if(c->c_box == 0 && x->b_ready_to_draw)
78  {
79  ebox_create_window(x, glist);
80  ebox_invalidate_all(x);
81  // No redraw for the 1st paint
82  if(x->b_obj.o_canvas && x->b_ready_to_draw && c->c_box == 0)
83  {
84  ebox_invalidate_layer(x, gensym("eboxbd"));
85  ebox_invalidate_layer(x, gensym("eboxio"));
86 
87  ebox_update(x);
88  if(c->c_widget.w_paint)
89  c->c_widget.w_paint(x, (t_object *)x->b_obj.o_canvas);
90  ebox_draw_border(x);
91  if(x->b_obj.o_canvas->gl_edit)
92  ebox_draw_iolets(x);
93  }
94  }
95  }
96  else
97  {
98  ebox_erase(x);
99  }
100  canvas_fixlinesfor(glist_getcanvas(glist), (t_text*)x);
101 }
