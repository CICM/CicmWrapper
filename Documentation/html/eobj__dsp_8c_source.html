37 void eobj_dspsetup(void *x, long nins, long nouts)
38 {
39  int i;
40  t_edspobj* z = (t_edspobj *)x;
41  nins = pd_clip_min(nins, 1);
42  nouts = pd_clip_min(nouts, 0);
43  z->d_perform_method = NULL;
44  z->d_misc = 1;
45  for( i = 0; i < 256; i++)
46  z->d_sigs_out[i] = (t_float *)calloc(8192, sizeof(t_float));
47  for(i = 1; i < nins; i++)
48  z->d_inlets[i] = signalinlet_new(&z->d_obj.o_obj, z->d_float);
49  for(i = 0; i < nouts; i++)
50  z->d_outlets[i] = outlet_new(&z->d_obj.o_obj, &s_signal);
51 
52  z->d_dsp_size = obj_nsiginlets(&z->d_obj.o_obj) + obj_nsigoutlets(&z->d_obj.o_obj) + 6;
53 }

202 t_int* eobj_perform(t_int* w)
203 {
204  int i, j;
205  t_edspobj* x = (t_edspobj *)(w[1]);
206  long nsamples = (long)(w[2]);
207  long flag = (long)(w[3]);
208  void* user_p = (void *)(w[4]);
209  long nins = (long)(w[5]);
210  long nouts = (long)(w[6]);
211  t_float** ins = (t_float **)(&w[7]);
212  t_float** outs = (t_float **)(&w[7 + nins]);
213 
214  if(x->d_misc == E_NO_INPLACE)
215  {
216  t_float *outs_real, *outs_perf;
217  if(x->d_perform_method != NULL)
218  x->d_perform_method(x, NULL, ins, nins, x->d_sigs_out, nouts, nsamples, flag, user_p);
219  for(i = 0; i < nouts; i++)
220  {
221  outs_perf = x->d_sigs_out[i];
222  outs_real = outs[i];
223  for(j = 0; j < nsamples; j++)
224  {
225  outs_real[j] = outs_perf[j];
226  }
227  }
228  }
229  else
230  {
231  if(x->d_perform_method != NULL)
232  x->d_perform_method(x, NULL, ins, nins, outs, nouts, nsamples, flag, user_p);
233  }
234  return w + (x->d_dsp_size + 1);
235 }
