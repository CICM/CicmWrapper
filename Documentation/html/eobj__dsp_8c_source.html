37 void eobj_dspsetup(void *x, long nins, long nouts)
38 {
39  int i;
40  t_edspobj* z = (t_edspobj *)x;
41  nins = pd_clip_min(nins, 1);
42  nouts = pd_clip_min(nouts, 0);
43  z->d_perform_method = NULL;
44  z->d_misc = 1;
45  for( i = 0; i < 256; i++)
46  z->d_sigs_out[i] = (t_float *)calloc(8192, sizeof(t_float));
47  for(i = 1; i < nins; i++)
48  z->d_inlets[i] = signalinlet_new(&z->d_obj.o_obj, z->d_float);
49  for(i = 0; i < nouts; i++)
50  z->d_outlets[i] = outlet_new(&z->d_obj.o_obj, &s_signal);
51 
52  z->d_dsp_size = obj_nsiginlets(&z->d_obj.o_obj) + obj_nsigoutlets(&z->d_obj.o_obj) + 6;
53 }

83  nins = pd_clip_min(nins, 1);
84 
85  if(nins > obj_nsiginlets(&z->d_obj.o_obj))
86  {
87  for(i = obj_nsiginlets(&z->d_obj.o_obj); i < nins; i++)
88  {
89  z->d_inlets[i] = signalinlet_new(&z->d_obj.o_obj, z->d_float);
90  }
91  }
92  else if(nins < obj_nsiginlets(&z->d_obj.o_obj))
93  {
94  for(i = obj_nsiginlets(&z->d_obj.o_obj) - 1; i >= nins - 1; i--)
95  {
96  canvas_deletelines_for_io(z->d_obj.o_canvas, (t_text *)x, z->d_inlets[i], NULL);
97  inlet_free(z->d_inlets[i]);
98  }
99  }
100  z->d_dsp_size = obj_nsiginlets(&z->d_obj.o_obj) + obj_nsigoutlets(&z->d_obj.o_obj) + 6;
101  if(!eobj_getregister(x))
102  ebox_redraw(x);
103 }

179 void eobj_dsp(t_edspobj *x, t_signal **sp)
180 {
181  int i;
182  t_eclass *c = (t_eclass *)x->d_obj.o_obj.te_g.g_pd;
183  short* count = (short*)calloc((obj_nsiginlets(&x->d_obj.o_obj) + obj_nsigoutlets(&x->d_obj.o_obj)), sizeof(short));
184 
185  if(c->c_widget.w_dsp != NULL)
186  c->c_widget.w_dsp(x, x, &count, sp[0]->s_sr, sp[0]->s_n, 0);
187 
188  x->d_dsp_vectors[0] = (t_int)x;
189  x->d_dsp_vectors[1] = (t_int)sp[0]->s_n;
190  x->d_dsp_vectors[2] = (t_int)x->d_dsp_flag;
191  x->d_dsp_vectors[3] = (t_int)x->d_dsp_user_param;
192  x->d_dsp_vectors[4] = (t_int)obj_nsiginlets(&x->d_obj.o_obj);
193  x->d_dsp_vectors[5] = (t_int)obj_nsigoutlets(&x->d_obj.o_obj);
194 
195  for(i = 6; i < x->d_dsp_size; i++)
196  {
197  x->d_dsp_vectors[i] = (t_int)(sp[i - 6]->s_vec);
198  }
199 
200  dsp_addv(eobj_perform, (int)x->d_dsp_size, x->d_dsp_vectors);
201  free(count);
202 }

210 t_int* eobj_perform(t_int* w)
211 {
212  int i, j;
213  t_edspobj* x = (t_edspobj *)(w[1]);
214  long nsamples = (long)(w[2]);
215  long flag = (long)(w[3]);
216  void* user_p = (void *)(w[4]);
217  long nins = (long)(w[5]);
218  long nouts = (long)(w[6]);
219  t_float** ins = (t_float **)(&w[7]);
220  t_float** outs = (t_float **)(&w[7 + nins]);
221 
222  if(x->d_misc == E_NO_INPLACE)
223  {
224  t_float *outs_real, *outs_perf;
225  if(x->d_perform_method != NULL)
226  x->d_perform_method(x, NULL, ins, nins, x->d_sigs_out, nouts, nsamples, flag, user_p);
227  for(i = 0; i < nouts; i++)
228  {
229  outs_perf = x->d_sigs_out[i];
230  outs_real = outs[i];
231  for(j = 0; j < nsamples; j++)
232  {
233  outs_real[j] = outs_perf[j];
234  }
235  }
236  }
237  else
238  {
239  if(x->d_perform_method != NULL)
240  x->d_perform_method(x, NULL, ins, nins, outs, nouts, nsamples, flag, user_p);
241  }
242  return w + (x->d_dsp_size + 1);
243 }
