69 /*
70  \ @memberof ebox
71  \ @param x The ebox pointer
72  \ @param name The layer name
73  \ @param width The width of the layer
74  \ @param height The height of the layer
75  \ @return Return an elayer or NULL if layer name is already used or if you can't write the layer
76 */
77 t_elayer* ebox_start_layer(t_ebox *x, t_symbol *name, float width, float height)
78 {
79  int i;
80  char text[256];
81 
82  for(i = 0; i < x->b_number_of_layers; i++)
83  {
84  t_elayer* graphic = &x->b_layers[i];
85  if(graphic->e_name == name)
86  {
87  if(graphic->e_state == EGRAPHICS_INVALID)
88  {
89  graphic->e_owner = (t_object *)x;
90 
91  egraphics_matrix_init(&graphic->e_matrix, 1., 0., 0., 1., 0., 0.);
92  graphic->e_line_width = 1.f;
93  graphic->e_color = gensym("#000000");
94  graphic->e_rect.x = 0.f;
95  graphic->e_rect.y = 0.f;
96  graphic->e_rect.height = pd_clip_min(height, 0.);
97  graphic->e_rect.width = pd_clip_min(width, 0.);
98 
99  graphic->e_number_objects = 0;
100  if(graphic->e_new_objects.e_points)
101  free(graphic->e_new_objects.e_points);
102  graphic->e_new_objects.e_points = NULL;
103  graphic->e_new_objects.e_npoints = 0;
104  graphic->e_new_objects.e_roundness = 0.;
105  graphic->e_objects = NULL;
106  sprintf(text, "%s%ld", name->s_name, (long)x);
107  graphic->e_id = gensym(text);
108 
109  graphic->e_state = EGRAPHICS_OPEN;
110  return &x->b_layers[i];
111  }
112  else
113  {
114  return NULL;
115  }
116  }
117  }
118  if(x->b_layers == NULL)
119  x->b_layers = (t_elayer*)calloc(1, sizeof(t_elayer));
120  else
121  x->b_layers = (t_elayer*)realloc(x->b_layers, (x->b_number_of_layers + 1) * sizeof(t_elayer));
122  if(x->b_layers)
123  {
124  t_elayer* graphic = x->b_layers+x->b_number_of_layers;
125  x->b_number_of_layers++;
126 
127  graphic->e_owner = (t_object *)x;
128 
129  egraphics_matrix_init(&graphic->e_matrix, 1., 0., 0., 1., 0., 0.);
130  graphic->e_line_width = 1.f;
131  graphic->e_color = gensym("#000000");
132  graphic->e_rect.x = 0.f;
133  graphic->e_rect.y = 0.f;
134  graphic->e_rect.height = pd_clip_min(height, 0.);
135  graphic->e_rect.width = pd_clip_min(width, 0.);
136 
137 
138  graphic->e_number_objects = 0;
139  graphic->e_new_objects.e_points = NULL;
140  graphic->e_new_objects.e_npoints = 0;
141  graphic->e_new_objects.e_roundness = 0.;
142  graphic->e_objects = NULL;
143 
144  graphic->e_state = EGRAPHICS_OPEN;
145  graphic->e_name = name;
146  sprintf(text, "%s%ld", name->s_name, (long)x);
147  graphic->e_id = gensym(text);
148  return graphic;
149  }
150  else
151  {
152  return NULL;
153  }
154 }

209 {
210  int i, j;
211  float bdsize, start, extent;
212  t_elayer* g = NULL;
213  bdsize = x->b_boxparameters.d_borderthickness;
214  sys_vgui("%s configure -bg %s\n", x->b_drawing_id->s_name, rgba_to_hex(x->b_boxparameters.d_boxfillcolor));
215 
216  for(i = 0; i < x->b_number_of_layers; i++)
217  {
218  if(x->b_layers[i].e_name == name)
219  {
220  g = &x->b_layers[i];
221  if(g->e_state != EGRAPHICS_TODRAW)
222  {
223  return -1;
224  }
225  }
226  }
227  if(g)
228  {
229 
230  for(i = 0; i < g->e_number_objects; i++)
231  {
232  t_egobj* gobj = g->e_objects+i;
234  if(gobj->e_type == E_GOBJ_PATH || gobj->e_type == E_GOBJ_RECT)
235  {
236  if(gobj->e_filled)
237  sys_vgui("%s create polygon ", x->b_drawing_id->s_name);
238  else
239  sys_vgui("%s create line ", x->b_drawing_id->s_name);
240 
241  for(j = 0; j < gobj->e_npoints; j ++)
242  sys_vgui("%d %d ", (int)(gobj->e_points[j].x + x_p + bdsize), (int)(gobj->e_points[j].y + y_p + bdsize));
243 
244  if(gobj->e_filled)
245  sys_vgui("-fill %s -width 0 -tags { %s %s }\n", gobj->e_color->s_name, g->e_id->s_name, x->b_all_id->s_name);
246  else
247  sys_vgui("-fill %s -width %f -tags { %s %s }\n", gobj->e_color->s_name, gobj->e_width, g->e_id->s_name, x->b_all_id->s_name);
248 
249  g->e_state = EGRAPHICS_CLOSE;
250  }
252  else if (gobj->e_type == E_GOBJ_OVAL)
253  {
254  sys_vgui("%s create oval %d %d %d %d ",
255  x->b_drawing_id->s_name,
256  (int)(gobj->e_points[0].x + x_p + bdsize),
257  (int)(gobj->e_points[0].y + y_p + bdsize),
258  (int)(gobj->e_points[1].x + x_p + bdsize),
259  (int)(gobj->e_points[1].y + y_p + bdsize));
260  if(gobj->e_filled)
261  sys_vgui("-fill %s -width 0 -tags { %s %s }\n", gobj->e_color->s_name, g->e_id->s_name, x->b_all_id->s_name);
262  else
263  sys_vgui("-outline %s -width %f -tags { %s %s }\n", gobj->e_color->s_name, gobj->e_width, g->e_id->s_name, x->b_all_id->s_name);
264 
265  g->e_state = EGRAPHICS_CLOSE;
266  }
268  else if (gobj->e_type == E_GOBJ_ARC)
269  {
270  start = pd_angle(gobj->e_points[2].x, gobj->e_points[2].y);
271  extent = pd_angle(gobj->e_points[3].x, gobj->e_points[3].y);
272  sys_vgui("%s create arc %d %d %d %d -start %f -extent %f ",
273  x->b_drawing_id->s_name,
274  (int)(gobj->e_points[0].x + x_p + bdsize),
275  (int)(gobj->e_points[0].y + y_p + bdsize),
276  (int)(gobj->e_points[1].x + x_p + bdsize),
277  (int)(gobj->e_points[1].y + y_p + bdsize),
278  (float)start / EPD_2PI * 360.,
279  (float)extent / EPD_2PI * 360.);
280 
281  if(gobj->e_filled)
282  sys_vgui("-style pieslice -fill %s -width 0 -tags { %s %s }\n", gobj->e_color->s_name, g->e_id->s_name, x->b_all_id->s_name);
283  else
284  sys_vgui("-style arc -outline %s -width %f -tags { %s %s }\n", gobj->e_color->s_name, gobj->e_width, g->e_id->s_name, x->b_all_id->s_name);
285 
286  g->e_state = EGRAPHICS_CLOSE;
287  }
289  else if(gobj->e_type == E_GOBJ_TEXT)
290  {
291 
292  sys_vgui("%s create text %d %d -text {%s} -anchor %s -justify %s -font {%s %d %s} -fill %s -width %d -tags { %s %s }\n",
293  x->b_drawing_id->s_name,
294  (int)(gobj->e_points[0].x + x_p + bdsize),
295  (int)(gobj->e_points[0].y + y_p + bdsize),
296  gobj->e_text->s_name,
297  gobj->e_anchor->s_name,
298  gobj->e_justify->s_name,
299  gobj->e_font.c_family->s_name, (int)gobj->e_font.c_size, gobj->e_font.c_weight->s_name,
300  gobj->e_color->s_name,
301  (int)(gobj->e_points[1].x),
302  g->e_id->s_name,
303  x->b_all_id->s_name);
304 
305  g->e_state = EGRAPHICS_CLOSE;
306  }
307  else if(gobj->e_type == E_GOBJ_IMAG)
308  {
309  /*
310  //sys_vgui("set %s [image create photo -file %s%s.%s]\n", image->c_name->s_name, path->s_name, name->s_name, image->c_ext->s_name);

318  g->e_state = EGRAPHICS_CLOSE;
319  }
320  else
321  {
322  return -1;
323  }
324  }
325  }
326  else
327  {
328  return -1;
329  }

343 void ebox_draw_border(t_ebox* x)
344 {
345  float bdsize, bdcorner;
346  t_elayer* g = NULL;
347  bdcorner = pd_clip_max(x->b_boxparameters.d_cornersize, x->b_boxparameters.d_borderthickness - 1);
348  bdsize = x->b_boxparameters.d_borderthickness;
349  g = ebox_start_layer(x, gensym("eboxbd"), x->b_rect.width, x->b_rect.height);
350 
351  if(g)
352  {
353  if(x->b_selected_box == EITEM_OBJ)
354  {
355  egraphics_set_color_rgba(g, &rgba_blue);
356  }
357  else
358  {
359  egraphics_set_color_rgba(g, &x->b_boxparameters.d_bordercolor);
360  }
361  egraphics_set_line_width(g, bdsize*2);
362  egraphics_rectangle_rounded(g, 0, 0, x->b_rect.width+bdsize*2, x->b_rect.height+bdsize*2, bdcorner);
363  egraphics_stroke(g);
364 
365  ebox_end_layer(x, gensym("eboxbd"));
366  }
367  ebox_paint_layer(x, gensym("eboxbd"), -bdsize, -bdsize);
368 }

376 void ebox_draw_iolets(t_ebox* x)
377 {
378  int i;
379  float bdsize;
380  t_elayer* g = NULL;
381  bdsize = x->b_boxparameters.d_borderthickness;
382  g = ebox_start_layer(x, gensym("eboxio"), x->b_rect.width, x->b_rect.height);
383 
384  if(g)
385  {
386  egraphics_set_line_width(g, 1);
387  for(i = 0; i < obj_ninlets((t_object *)x); i++)
388  {
389  int pos_x_inlet = 0;
390  if(obj_ninlets((t_object *)x) != 1)
391  pos_x_inlet = (int)(i / (float)(obj_ninlets((t_object *)x) - 1) * (x->b_rect.width - 8));
392  egraphics_rectangle(g, pos_x_inlet, 0, 7, 2);
393  if(x->b_selected_inlet == i)
394  {
395  egraphics_set_color_rgba(g, &rgba_blue);
396  egraphics_fill(g);
397  }
398  else if (obj_issignalinlet((t_object *)x, i))
399  {
400  egraphics_set_color_rgba(g, &rgba_inletsig);
401  egraphics_fill(g);
402  }
403  else if(obj_isfloatinlet((t_object *)x, i))
404  {
405  egraphics_set_color_rgba(g, &rgba_black);
406  egraphics_fill(g);
407  }
408  else
409  {
410  egraphics_set_color_rgba(g, &rgba_white);
411  egraphics_fill(g);
412  egraphics_set_color_rgba(g, &rgba_black);
413  egraphics_stroke(g);
414  }
415  }
416 
417  for(i = 0; i < obj_noutlets((t_object *)x); i++)
418  {
419  int pos_x_outlet = 0;
420  if(obj_noutlets((t_object *)x) != 1)
421  pos_x_outlet = (int)(i / (float)(obj_noutlets((t_object *)x) - 1) * (x->b_rect.width - 8));
422  egraphics_rectangle(g, pos_x_outlet, x->b_rect.height - 3 + bdsize*2, 7, 2);
423  if(x->b_selected_outlet == i)
424  {
425  egraphics_set_color_rgba(g, &rgba_blue);
426  egraphics_fill(g);
427  }
428  else if (obj_issignaloutlet((t_object *)x, i))
429  {
430  egraphics_set_color_rgba(g, &rgba_inletsig);
431  egraphics_fill(g);
432  }
433  else if(obj_isfloatoutlet((t_object *)x, i))
434  {
435  egraphics_set_color_rgba(g, &rgba_black);
436  egraphics_fill(g);
437  }
438  else
439  {
440  egraphics_set_color_rgba(g, &rgba_white);
441  egraphics_fill(g);
442  egraphics_set_color_rgba(g, &rgba_black);
443  egraphics_stroke(g);
444  }
445  }
446  ebox_end_layer(x, gensym("eboxio"));
447  }
448  ebox_paint_layer(x, gensym("eboxio"), 0, -bdsize);
449 }
