220 void egraphics_arc_oval(t_elayer *g, float xc, float yc, float radiusx, float radiusy, float angle1, float angle2)
221 {
222  while (angle1 >= EPD_2PI)
223  angle1 -= EPD_2PI;
224  while (angle1 < 0)
225  angle1 += EPD_2PI;
226 
227  while (angle2 >= EPD_2PI)
228  angle2 -= EPD_2PI;
229  while (angle2 < 0)
230  angle2 += EPD_2PI;
231 
232  if((angle1 - angle2) < 0.00001f && (angle1 - angle2) > -0.00001f)
233  {
234  egraphics_oval(g, xc, yc, radiusx, radiusy);
235  return;
236  }
237  if(g->e_state == EGRAPHICS_OPEN)
238  {
239  if(g->e_new_objects.e_points == NULL)
240  g->e_new_objects.e_points = (t_pt *)calloc(4, sizeof(t_pt));
241  else
242  g->e_new_objects.e_points = (t_pt *)realloc(g->e_new_objects.e_points , 4 * sizeof(t_pt));
243  if(g->e_new_objects.e_points)
244  {
245  g->e_new_objects.e_type = E_GOBJ_ARC;
246  g->e_new_objects.e_points[0].x = xc - radiusx;
247  g->e_new_objects.e_points[0].y = yc - radiusy;
248  g->e_new_objects.e_points[1].x = xc + radiusx;
249  g->e_new_objects.e_points[1].y = yc + radiusy;
250  g->e_new_objects.e_points[2].x = pd_abscissa(radiusx, angle1);
251  g->e_new_objects.e_points[2].y = pd_ordinate(radiusy, angle1);
252  g->e_new_objects.e_points[3].x = pd_abscissa(radiusx, angle2-angle1);
253  g->e_new_objects.e_points[3].y = pd_ordinate(radiusy, angle2-angle1);
254  g->e_new_objects.e_npoints = 4;
255  }
256  else
257  {
258  g->e_new_objects.e_type = E_GOBJ_INVALID;
259  }
260  }
