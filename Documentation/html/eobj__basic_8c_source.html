36 void *eobj_new(t_eclass *c)
37 {
38  t_pd* x;
39  t_eobj* z;
40  char buffer[MAXPDSTRING];
41  if(!c)
42  bug("pd_new: apparently called before setup routine");
43  x = (t_pd *)t_getbytes(c->c_class.c_size);
44  *x = (t_pd)c;
45 
46  if (c->c_class.c_patchable)
47  {
48  ((t_object *)x)->ob_inlet = 0;
49  ((t_object *)x)->ob_outlet = 0;
50  }
51  z = (t_eobj *)x;
52  z->o_nproxy = 0;
53  z->o_canvas = canvas_getcurrent();
54  sprintf(buffer,"#%s%lx", c->c_class.c_name->s_name, (long unsigned int)z);
55  z->o_id = gensym(buffer);
56  pd_bind(&z->o_obj.ob_pd, z->o_id);
57  eobj_attach_torouter((t_object *)z);
58  sprintf(buffer,".x%lx.c", (long unsigned int)z->o_canvas);
59  z->o_canvas_id = gensym(buffer);
60  c->c_widget.w_dosave = (method)eobj_dosave;
61  return (x);
62 }
