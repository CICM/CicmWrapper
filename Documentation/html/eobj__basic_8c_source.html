35 void *eobj_alloc(t_eclass *c)
36 {
37  t_pd* x;
38  t_eobj* z;
39  if(!c)
40  bug("pd_new: apparently called before setup routine");
41  x = (t_pd *)t_getbytes(c->c_class.c_size);
42  *x = (t_pd)c;
43 
44  if (c->c_class.c_patchable)
45  {
46  ((t_object *)x)->ob_inlet = 0;
47  ((t_object *)x)->ob_outlet = 0;
48  }
49  z = (t_eobj *)x;
50  z->o_nproxy = 0;
51 
52  return (x);
53 }
