37 void ebox_dspsetup(t_dspobj *x, long nins, long nouts)
38 {
39  int i;
40  nins = pd_clip_min(nins, 1);
41  nouts = pd_clip_min(nouts, 0);
42  x->d_perform_method = NULL;
43  x->d_misc = 1;
44  for( i = 0; i < 256; i++)
45  x->d_sigs_out[i] = (t_float *)calloc(8192, sizeof(t_float));
46  for(i = 1; i < nins; i++)
47  x->d_inlets[i] = signalinlet_new(&x->d_obj.o_obj, x->d_float);
48  for(i = 0; i < nouts; i++)
49  x->d_outlets[i] = outlet_new(&x->d_obj.o_obj, &s_signal);
50 
51  x->d_dsp_size = obj_nsiginlets(&x->d_obj.o_obj) + obj_nsigoutlets(&x->d_obj.o_obj) + 6;
52 }
