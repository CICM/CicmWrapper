97 void default_attr_process(void *x)
98 {
99  int i, j, k;
100  long defc = 0;
101  t_atom* defv = NULL;
102  t_ebox* z = (t_ebox *)x;
103  t_eclass* c = (t_eclass *)z->b_obj.o_obj.te_g.g_pd;
104 
105  for(i = 0; i < c->c_nattr; i++)
106  {
107  if(c->c_attr[i].defvals)
108  {
109  defc = c->c_attr[i].size;
110  defv = (t_atom *)calloc(defc, sizeof(t_atom));
111  if(defc && defv)
112  {
113  char check = 0;
114  char* str_start = c->c_attr[i].defvals->s_name;
115  for(j = 0; j < defc; j++)
116  {
117  for(k = 0; k < strlen(str_start); k++)
118  {
119  if(isalpha(str_start[k]))
120  check = 1;
121  }
122  if(check || strpbrk(str_start, "<>()'\""))
123  {
124  atom_setsym(defv+j, gensym(str_start));
125  }
126  else
127  {
128  float val = (float)strtod(str_start, &str_start);
129  atom_setfloat(defv+j, val);
130  }
131  }
132  object_attr_setvalueof((t_object *)x, c->c_attr[i].name, defc, defv);
133  defc = 0;
134  free(defv);
135  defv = NULL;
136  }
137  }
138  }

222 /*
223  \ @memberof ebox
224  \ @param x The ebox
225  \ @param attr Nothing (for Max 6 compatibility)
226  \ @param argc The size of the array of atoms
227  \ @param argv The array of atoms that contains the new width and the new height
228  \ @return Always 0 (for the moment)
229 */
230 t_pd_err ebox_size_set(t_ebox *x, t_object *attr, long argc, t_atom *argv)
231 {
232  float width, height;
233  if(argc && argv)
234  {
235  if(x->b_flags & EBOX_GROWNO)
236  return 0;
237  else if(x->b_flags & EBOX_GROWLINK)
238  {
239  if(atom_gettype(argv) == A_FLOAT)
240  {
241  width = pd_clip_min(atom_getfloat(argv), 4);
242  height = x->b_rect.height;
243  x->b_rect.height += width - x->b_rect.width;
244  if(x->b_rect.height < 4)
245  {
246  x->b_rect.width += 4 - height;
247  x->b_rect.height = 4;
248  }
249  else
250  {
251  x->b_rect.width = width;
252  }
253  }
254  }
255  else if (x->b_flags & EBOX_GROWINDI)
256  {
257  if(atom_gettype(argv) == A_FLOAT)
258  x->b_rect.width = pd_clip_min(atom_getfloat(argv), 4);
259  if(atom_gettype(argv+1) == A_FLOAT)
260  x->b_rect.height = pd_clip_min(atom_getfloat(argv+1), 4);
261  }
262  }
263 
264  return 0;
265 }
