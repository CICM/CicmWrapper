177 void ebox_mouse_move(t_ebox* x, t_symbol* s, long argc, t_atom* argv)
178 {
179  t_atom av[2];
180  t_eclass *c = (t_eclass *)x->b_obj.o_obj.te_g.g_pd;
181  ebox_get_mouse_global_position(x);
182 
183  x->b_modifiers = (long)atom_getfloat(argv+2);
184 #ifdef _WINDOWS
185 
186  if(x->b_modifiers >= 131080)
187  {
188  x->b_modifiers -= 131080;
189  x->b_modifiers += EMOD_ALT;
190  }
191  else
192  x->b_modifiers -= 8;
193 #endif
194 
195  if(!x->b_obj.o_canvas->gl_edit)
196  {
197  if(!x->b_mouse_down)
198  {
199  if(x->b_flags & EBOX_IGNORELOCKCLICK)
200  ebox_set_cursor(x, 0);
201  else
202  ebox_set_cursor(x, 1);
203 
204  x->b_mouse.x = atom_getfloat(argv);
205  x->b_mouse.y = atom_getfloat(argv+1);
206 
207  if(c->c_widget.w_mousemove)
208  c->c_widget.w_mousemove(x, x->b_obj.o_canvas, x->b_mouse, x->b_modifiers);
209  }
210  else
211  {
212  ebox_mouse_drag(x, s, argc, argv);
213  }
214  }
215  else
216  {
217  if(!x->b_mouse_down)
218  {
219  ebox_mouse_move_editmode(x, atom_getfloat(argv), atom_getfloat(argv+1), x->b_modifiers);
220  }
221  else
222  {
223  x->b_mouse.x = atom_getfloat(argv);
224  x->b_mouse.y = atom_getfloat(argv+1);
225  if(x->b_selected_item == EITEM_NONE)
226  {
227  x->b_move_box = ebox_get_mouse_canvas_position(x);
228  sys_vgui("pdtk_canvas_motion %s %i %i 0\n", x->b_canvas_id->s_name, (int)x->b_move_box.x, (int)x->b_move_box.y);
229  }
230  else
231  {
232  if(x->b_flags & EBOX_GROWNO)
233  return;
234  else if(x->b_flags & EBOX_GROWLINK)
235  {
236  if(x->b_selected_item == EITEM_BOTTOM)
237  {
238  atom_setfloat(av, x->b_rect_last.width + (x->b_mouse.y - x->b_rect_last.height));
239  atom_setfloat(av+1, x->b_mouse.y);
240  }
241  else if(x->b_selected_item == EITEM_RIGHT)
242  {
243  atom_setfloat(av, x->b_mouse.x);
244  atom_setfloat(av+1, x->b_rect_last.height + (x->b_mouse.x - x->b_rect_last.width));
245  }
246  else if(x->b_selected_item == EITEM_CORNER)
247  {
248  if(x->b_mouse.y > x->b_mouse.x)
249  {
250  atom_setfloat(av, x->b_mouse.y);
251  atom_setfloat(av+1, x->b_mouse.y);
252  }
253  else
254  {
255  atom_setfloat(av, x->b_mouse.x);
256  atom_setfloat(av+1, x->b_mouse.x);
257  }
258  }
259  }
260  else if (x->b_flags & EBOX_GROWINDI)
261  {
262  if(x->b_selected_item == EITEM_BOTTOM)
263  {
264  atom_setfloat(av, x->b_rect_last.width);
265  atom_setfloat(av+1, x->b_mouse.y);
266  }
267  else if(x->b_selected_item == EITEM_RIGHT)
268  {
269  atom_setfloat(av, x->b_mouse.x);
270  atom_setfloat(av+1, x->b_rect_last.height);
271  }
272  else if(x->b_selected_item == EITEM_CORNER)
273  {
274  atom_setfloat(av, x->b_mouse.x);
275  atom_setfloat(av+1, x->b_mouse.y);
276  }
277  }
278  mess3((t_pd *)x, gensym("size"), gensym("size"), 2, av);
279  }
280  }
281  }
282 }

479  \ @memberof ebox

481  \ @param s The message selector
482  \ @param argc The size of the array of atoms
483  \ @param argv The array of atoms
484  \ @return Nothing
485 */
486 void ebox_mouse_move_editmode(t_ebox* x, float x_p, float y_p, float key)
487 {
488  int i;
489  int right, bottom;
490 
491  x->b_selected_outlet = -1;
492  x->b_selected_inlet = -1;
493  x->b_selected_item = EITEM_NONE;
494  x->b_move_box = ebox_get_mouse_canvas_position(x);
495  sys_vgui("pdtk_canvas_motion %s %i %i 0\n", x->b_canvas_id->s_name, (int)x->b_move_box.x, (int)x->b_move_box.y);
496 
497  right = x->b_rect.width + x->b_boxparameters.d_borderthickness * 2.;
498  bottom = x->b_rect.height + x->b_boxparameters.d_borderthickness * 2.;
499 
500  // TOP //
501  if(y_p >= 0 && y_p < 3)
502  {
503  for(i = 0; i < obj_noutlets((t_object *)x); i++)
504  {
505  int pos_x_inlet = 0;
506  if(obj_ninlets((t_object *)x) != 1)
507  pos_x_inlet = (int)(i / (float)(obj_ninlets((t_object *)x) - 1) * (x->b_rect.width - 8));
508 
509  if(x_p >= pos_x_inlet && x_p <= pos_x_inlet +7)
510  {
511  x->b_selected_inlet = i;
512  ebox_set_cursor(x, 4);
513  break;
514  }
515  }
516  ebox_invalidate_layer(x, gensym("eboxio"));
517  ebox_redraw(x);
518  return;
519  }
520  // BOTTOM & RIGHT //
521  else if(y_p > bottom - 3 && y_p <= bottom && x_p > right - 3 && x_p <= right)
522  {
523  x->b_selected_item = EITEM_CORNER;
524  ebox_set_cursor(x, 8);
525  return;
526  }
527  // BOTTOM //
528  else if(y_p > bottom - 3 && y_p < bottom)
529  {
530  for(i = 0; i < obj_noutlets((t_object *)x); i++)
531  {
532  int pos_x_outlet = 0;
533  if(obj_noutlets((t_object *)x) != 1)
534  pos_x_outlet = (int)(i / (float)(obj_noutlets((t_object *)x) - 1) * (x->b_rect.width - 8));
535 
536  if(x_p >= pos_x_outlet && x_p <= pos_x_outlet +7)
537  {
538  x->b_selected_outlet = i;
539  ebox_set_cursor(x, 5);
540  break;
541  }
542  }
543  if(x->b_selected_outlet == -1)
544  {
545  x->b_selected_item = EITEM_BOTTOM;
546  ebox_set_cursor(x, 7);
547  }
548  ebox_invalidate_layer(x, gensym("eboxio"));
549  ebox_redraw(x);
550  return;
551  }
552  // RIGHT //
553  else if(x_p > right - 3 && x_p <= right)
554  {
555  x->b_selected_item = EITEM_RIGHT;
556  ebox_set_cursor(x, 9);
557  return;
558  }
559 
560  // BOX //
561  ebox_set_cursor(x, 4);
562 
563  ebox_invalidate_layer(x, gensym("eboxio"));
564  ebox_redraw(x);
565 }
