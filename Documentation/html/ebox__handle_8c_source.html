102 void ebox_mouse_move(t_ebox* x, t_symbol* s, long argc, t_atom* argv)
103 {
104  t_atom av[2];
105  t_eclass *c = eobj_getclass(x);
106  eobj_get_mouse_global_position(x);
107 
108  x->b_modifiers = modifier_wrapper((long)atom_getfloat(argv+2));
109 
110 
111  if(!x->b_obj.o_canvas->gl_edit)
112  {
113  if(!x->b_mouse_down)
114  {
115  if(x->b_flags & EBOX_IGNORELOCKCLICK)
116  ebox_set_cursor(x, 0);
117  else
118  ebox_set_cursor(x, 1);
119 
120  x->b_mouse.x = atom_getfloat(argv);
121  x->b_mouse.y = atom_getfloat(argv+1);
122 
123  if(c->c_widget.w_mousemove)
124  c->c_widget.w_mousemove(x, x->b_obj.o_canvas, x->b_mouse, x->b_modifiers);
125  }
126  else
127  {
128  ebox_mouse_drag(x, s, argc, argv);
129  }
130  }
131  else if(!x->b_isinsubcanvas)
132  {
133  if(!x->b_mouse_down)
134  {
135  ebox_mouse_move_editmode(x, atom_getfloat(argv), atom_getfloat(argv+1), x->b_modifiers);
136  }
137  else
138  {
139  x->b_mouse.x = atom_getfloat(argv);
140  x->b_mouse.y = atom_getfloat(argv+1);
141  if(x->b_selected_item == EITEM_NONE)
142  {
143  x->b_move_box = eobj_get_mouse_canvas_position(x);
144  sys_vgui("pdtk_canvas_motion %s %i %i 0\n", x->b_canvas_id->s_name, (int)x->b_move_box.x, (int)x->b_move_box.y);
145  }
146  else
147  {
148  if(x->b_flags & EBOX_GROWNO)
149  return;
150  else if(x->b_flags & EBOX_GROWLINK)
151  {
152  if(x->b_selected_item == EITEM_BOTTOM)
153  {
154  atom_setfloat(av, x->b_rect_last.width + (x->b_mouse.y - x->b_rect_last.height));
155  atom_setfloat(av+1, x->b_mouse.y);
156  }
157  else if(x->b_selected_item == EITEM_RIGHT)
158  {
159  atom_setfloat(av, x->b_mouse.x);
160  atom_setfloat(av+1, x->b_rect_last.height + (x->b_mouse.x - x->b_rect_last.width));
161  }
162  else if(x->b_selected_item == EITEM_CORNER)
163  {
164  if(x->b_mouse.y > x->b_mouse.x)
165  {
166  atom_setfloat(av, x->b_mouse.y);
167  atom_setfloat(av+1, x->b_mouse.y);
168  }
169  else
170  {
171  atom_setfloat(av, x->b_mouse.x);
172  atom_setfloat(av+1, x->b_mouse.x);
173  }
174  }
175  }
176  else if (x->b_flags & EBOX_GROWINDI)
177  {
178  if(x->b_selected_item == EITEM_BOTTOM)
179  {
180  atom_setfloat(av, x->b_rect_last.width);
181  atom_setfloat(av+1, x->b_mouse.y);
182  }
183  else if(x->b_selected_item == EITEM_RIGHT)
184  {
185  atom_setfloat(av, x->b_mouse.x);
186  atom_setfloat(av+1, x->b_rect_last.height);
187  }
188  else if(x->b_selected_item == EITEM_CORNER)
189  {
190  atom_setfloat(av, x->b_mouse.x);
191  atom_setfloat(av+1, x->b_mouse.y);
192  }
193  }
194  mess3((t_pd *)x, gensym("size"), gensym("size"), 2, av);
195  }
196  }
197  }
198 }

361 void ebox_mouse_move_editmode(t_ebox* x, float x_p, float y_p, float key)
362 {
363  int i;
364  int right, bottom;
365 
366  x->b_selected_outlet = -1;
367  x->b_selected_inlet = -1;
368  x->b_selected_item = EITEM_NONE;
369  x->b_move_box = eobj_get_mouse_canvas_position(x);
370  sys_vgui("pdtk_canvas_motion %s %i %i 0\n", x->b_canvas_id->s_name, (int)x->b_move_box.x, (int)x->b_move_box.y);
371 
372  right = x->b_rect.width + x->b_boxparameters.d_borderthickness * 2.;
373  bottom = x->b_rect.height + x->b_boxparameters.d_borderthickness * 2.;
374 
375  // TOP //
376  if(y_p >= 0 && y_p < 3)
377  {
378  for(i = 0; i < obj_noutlets((t_object *)x); i++)
379  {
380  int pos_x_inlet = 0;
381  if(obj_ninlets((t_object *)x) != 1)
382  pos_x_inlet = (int)(i / (float)(obj_ninlets((t_object *)x) - 1) * (x->b_rect.width - 8));
383 
384  if(x_p >= pos_x_inlet && x_p <= pos_x_inlet +7)
385  {
386  x->b_selected_inlet = i;
387  ebox_set_cursor(x, 4);
388  break;
389  }
390  }
391  ebox_invalidate_layer(x, gensym("eboxio"));
392  ebox_redraw(x);
393  return;
394  }
395  // BOTTOM & RIGHT //
396  else if(y_p > bottom - 3 && y_p <= bottom && x_p > right - 3 && x_p <= right)
397  {
398  x->b_selected_item = EITEM_CORNER;
399  ebox_set_cursor(x, 8);
400  return;
401  }
402  // BOTTOM //
403  else if(y_p > bottom - 3 && y_p < bottom)
404  {
405  for(i = 0; i < obj_noutlets((t_object *)x); i++)
406  {
407  int pos_x_outlet = 0;
408  if(obj_noutlets((t_object *)x) != 1)
409  pos_x_outlet = (int)(i / (float)(obj_noutlets((t_object *)x) - 1) * (x->b_rect.width - 8));
410 
411  if(x_p >= pos_x_outlet && x_p <= pos_x_outlet +7)
412  {
413  x->b_selected_outlet = i;
414  ebox_set_cursor(x, 5);
415  break;
416  }
417  }
418  if(x->b_selected_outlet == -1)
419  {
420  x->b_selected_item = EITEM_BOTTOM;
421  ebox_set_cursor(x, 7);
422  }
423  ebox_invalidate_layer(x, gensym("eboxio"));
424  ebox_redraw(x);
425  return;
426  }
427  // RIGHT //
428  else if(x_p > right - 3 && x_p <= right)
429  {
430  x->b_selected_item = EITEM_RIGHT;
431  ebox_set_cursor(x, 9);
432  return;
433  }
434 
435  // BOX //
436  ebox_set_cursor(x, 4);
437 
438  ebox_invalidate_layer(x, gensym("eboxio"));
439  ebox_redraw(x);
440 }
