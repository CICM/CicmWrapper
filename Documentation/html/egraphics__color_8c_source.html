97 t_hsla rgba_to_hsla(t_rgba color)
98 {
99  t_hsla ncolor;
100  float delta, deltar, deltag, deltab;
101  float max = color.red;
102  float min = color.red;
103  if(min > color.green)
104  min = color.green;
105  if(min > color.blue)
106  min = color.blue;
107  if(max < color.green)
108  max = color.green;
109  if(max < color.blue)
110  max = color.blue;
111  delta = max - min;
112  ncolor.alpha = color.alpha;
113  ncolor.lightness = (max + min) / 2.;
114  if(max == 0)
115  {
116  ncolor.hue = 0;
117  ncolor.saturation = 0;
118  }
119  else
120  {
121  if(ncolor.lightness < 0.5)
122  ncolor.saturation = delta / (max + min);
123  else
124  ncolor.saturation = delta / (2. - max - min);
125 
126  deltar = (((max - color.red ) / 6 ) + (delta / 2)) / delta;
127  deltag = (((max - color.green ) / 6 ) + (delta / 2)) / delta;
128  deltab = (((max - color.blue ) / 6 ) + (delta / 2)) / delta;
129 
130  if(color.red == max)
131  ncolor.hue = deltab - deltag;
132  else if(color.green == max)
133  ncolor.hue = (1. / 3.) + deltar - deltab;
134  else if(color.blue == max)
135  ncolor.hue = (2. / 3.) + deltag - deltar;

194 float Hue_2_RGB(float v1,float v2,float vH)
195 {
196  if(vH < 0.)
197  vH += 1.;
198  if(vH > 1.)
199  vH -= 1.;
200  if((6. * vH) < 1.)
201  return (v1 + (v2 - v1) * 6. * vH);
202  if((2. * vH) < 1.)
203  return v2;
204  if(( 3 * vH) < 2.)
205  return (v1 + (v2 - v1) * ((2. / 3.) - vH) * 6.);
206  return v1;

209 t_rgb hsl_to_rgb(t_hsl color)
210 {
211  float var1, var2;
212  t_rgb ncolor;
213  if(color.saturation == 0.)
214  {
215  ncolor.red = color.lightness;
216  ncolor.green = color.lightness;
217  ncolor.blue = color.lightness;
218  }
219  else
220  {
221  if(color.lightness < 0.5)
222  var2 = color.lightness * (1. + color.saturation);
223  else
224  var2 = (color.lightness + color.saturation) - (color.saturation * color.lightness);
225 
226  var1 = 2 * color.lightness - var2;
227 
228  ncolor.red = Hue_2_RGB(var1, var2, color.hue + (1. / 3.));
229  ncolor.green = Hue_2_RGB(var1, var2, color.hue);
230  ncolor.blue = Hue_2_RGB(var1, var2, color.hue - (1. / 3.));
